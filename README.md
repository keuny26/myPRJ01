### 1. 자료 구조:
import time

def run_mbti_test():
    """12가지 질문으로 구성된 MBTI 간이 검사를 실행하고 결과를 출력합니다."""

    # 4가지 MBTI 지표를 위한 점수 초기화
    # E-I (외향-내향), S-N (감각-직관), T-F (사고-감정), J-P (판단-인식)
    scores = {'E': 0, 'I': 0, 'S': 0, 'N': 0, 'T': 0, 'F': 0, 'J': 0, 'P': 0}

    # 질문 리스트: (질문, 긍정 답변(1)에 해당하는 지표, 부정 답변(2)에 해당하는 지표)
    # 각 지표에 3문항씩 할당했습니다.
    questions = [
        ("1. 처음 만나는 사람들과 이야기하는 것이 즐겁다.", 'E', 'I'), # E/I
        ("2. 새로운 아이디어나 가능성에 대해 공상하는 것을 좋아한다.", 'N', 'S'), # S/N
        ("3. 결정을 내릴 때 논리적이고 객관적인 분석을 우선한다.", 'T', 'F'), # T/F
        ("4. 할 일을 미리 계획하고 체계적으로 진행하는 것을 선호한다.", 'J', 'P'), # J/P

        ("5. 혼자만의 시간보다 다른 사람들과 함께하는 시간을 더 중요하게 생각한다.", 'E', 'I'), # E/I
        ("6. 현재의 사실이나 경험을 바탕으로 현실적인 해결책을 찾는 편이다.", 'S', 'N'), # S/N
        ("7. 다른 사람의 감정을 공감하고 배려하는 것이 분석보다 더 중요하다고 느낀다.", 'F', 'T'), # T/F
        ("8. 유동적이고 즉흥적인 상황에 더 잘 적응하고 편안함을 느낀다.", 'P', 'J'), # J/P

        ("9. 파티나 모임에서 에너지를 얻고 활력을 되찾는 편이다.", 'E', 'I'), # E/I
        ("10. 미래에 대한 비전이나 예측에 관심이 많다.", 'N', 'S'), # S/N
        ("11. 비판이나 지적을 할 때, 솔직함이 친절함보다 중요하다고 생각한다.", 'T', 'F'), # T/F
        ("12. 마감 기한 직전에 집중해서 일을 처리하는 것을 선호한다.", 'P', 'J'), # J/P
    ]

    print("--- 🧠 MBTI 간이 검사 시작 (12문항) 🧠 ---")
    print("각 문항에 대해 '1' 또는 '2'를 입력해 주세요.")
    print("------------------------------------------")
    time.sleep(1)

    for i, (question, positive_trait, negative_trait) in enumerate(questions):
        while True:
            try:
                # 사용자 입력 받기
                answer = input(f"\n{question}\n(1: 매우 그렇다 / 2: 그렇지 않다) -> ")

                if answer == '1':
                    scores[positive_trait] += 1
                    break
                elif answer == '2':
                    scores[negative_trait] += 1
                    break
                else:
                    print("⚠️ 잘못된 입력입니다. '1' 또는 '2'만 입력해주세요.")
            except EOFError:
                # 입력 종료 방지
                print("\n입력 오류가 발생했습니다. 다시 시도해주세요.")


    # 최종 MBTI 지표 결정
    # E/I 지표 결정 (E 점수가 I 점수보다 높으면 E, 아니면 I)
    e_i = 'E' if scores['E'] >= scores['I'] else 'I'
    s_n = 'S' if scores['S'] >= scores['N'] else 'N'
    t_f = 'T' if scores['T'] >= scores['F'] else 'F'
    j_p = 'J' if scores['J'] >= scores['P'] else 'P'

    final_mbti = e_i + s_n + t_f + j_p

    # 결과 출력
    print("\n\n=============== 🌟 검사 결과 🌟 ===============")
    print(f"당신의 간이 MBTI 유형은: **{final_mbti}** 입니다.")
    print("==================================================")
    
    # 해설 제공
    provide_explanation(final_mbti)
    
    # 각 지표별 점수 (선택 사항)
    # print("\n[점수 현황]")
    # print(f"E: {scores['E']}, I: {scores['I']} | S: {scores['S']}, N: {scores['N']} | T: {scores['T']}, F: {scores['F']} | J: {scores['J']}, P: {scores['P']}")
